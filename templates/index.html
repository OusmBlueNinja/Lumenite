-- RESTful API with JSON responses local users = {} -- Create user endpoint app:post("/api/users", function(req) local user_data = req.json if not user_data.email or not user_data.name then return json_error("Missing required fields", 400) end local user = { id = generate_id(), name = user_data.name, email = user_data.email, created_at = os.time() } users[user.id] = user return json_response(user, 201) end) -- Get all users app:get("/api/users", function(req) return json_response(table_values(users)) end) app:listen(5800)
